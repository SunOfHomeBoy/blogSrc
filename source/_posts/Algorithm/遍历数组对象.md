## 遍历数组对象

````
/** 参数说明
 * arr: array = 被遍历 数组proto
 * objKey: string = filter
 * key: string = 当前筛选项
 */
getIndex = function (arr, objKey, key) {
  for (let i = 0; i < arr.length; i++) {
    if (arr[i][objKey] === key) {
      return i;
    }
  }
  return -1;
}
````
